/*
 * Smart Fan Control System — ESP32
 * For People with Visual and Mobility Challenges
 *
 * Hardware:
 *   - ESP32 DevKit
 *   - DHT11 sensor (pin 4) — temperature & humidity
 *   - L298N Motor Driver:
 *       ESP32 D15 → ENA (PWM speed control)
 *       ESP32 D2  → IN2 (fan on/off direction)
 *       IN1 → GND or leave LOW (single direction for fan)
 *   - Buzzer (pin 27) — audio feedback
 *   - LCD 1602 I2C (SDA → D21, SCL → D22, VCC → 5V recommended)
 */

#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <DHT.h>
#include <Wire.h>
#include <hd44780.h>
#include <hd44780ioClass/hd44780_I2Cexp.h>

// ═══════════════════════════════════════════════════
// CONFIGURATION
// ═══════════════════════════════════════════════════
const char* WIFI_SSID      = "Uy Heang";
const char* WIFI_PASSWORD   = "011995596";
const char* SERVER_URL      = "http://192.168.100.41:8000";
const char* DEVICE_ID       = "ESP32-FAN-001";

// ═══════════════════════════════════════════════════
// HARDWARE PINS
// ═══════════════════════════════════════════════════
#define DHT_PIN       4
#define DHT_TYPE      DHT11           // ✅ changed back to DHT11
#define MOTOR_ENA     15
#define MOTOR_IN2     2
#define BUZZER_PIN    27

// I2C pins for LCD
#define I2C_SDA       21
#define I2C_SCL       22

// LCD I2C address (common: 0x27 or 0x3F)
#define LCD_ADDRESS   0x27
#define LCD_COLUMNS   16
#define LCD_ROWS      2

// PWM
#define PWM_FREQ       25000
#define PWM_RESOLUTION 8

// ═══════════════════════════════════════════════════
// TIMING
// ═══════════════════════════════════════════════════
#define POLL_INTERVAL       3000
#define SENSOR_INTERVAL     10000

// ═══════════════════════════════════════════════════
// GLOBALS
// ═══════════════════════════════════════════════════
DHT dht(DHT_PIN, DHT_TYPE);
hd44780_I2Cexp lcd(LCD_ADDRESS);

bool fanState = false;
int  fanSpeed = 255;
float lastTemperature = 0;
float lastHumidity = 0;

unsigned long lastPollTime = 0;
unsigned long lastSensorTime = 0;

// ═══════════════════════════════════════════════════
// SETUP
// ═══════════════════════════════════════════════════
void setup() {
    Serial.begin(115200);
    Serial.println("\n=== Smart Fan Control System ===");

    Wire.begin(I2C_SDA, I2C_SCL);

    // 🔍 Uncomment the next line to scan I2C devices
    // scanI2C();

    int lcdStatus = lcd.begin(LCD_COLUMNS, LCD_ROWS);
    if (lcdStatus) {
        Serial.print("❌ LCD init failed, error = ");
        Serial.println(lcdStatus);
    } else {
        lcd.backlight();
        lcd.clear();
        lcd.setCursor(0, 0);
        lcd.print("System Starting");
        lcd.setCursor(0, 1);
        lcd.print("Please wait...");
    }

    pinMode(MOTOR_IN2, OUTPUT);
    pinMode(BUZZER_PIN, OUTPUT);
    digitalWrite(MOTOR_IN2, LOW);

    ledcAttach(MOTOR_ENA, PWM_FREQ, PWM_RESOLUTION);
    ledcWrite(MOTOR_ENA, 0);

    dht.begin();

    beep(2, 100);
    connectWiFi();
    beep(1, 300);

    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("System Ready");
    delay(1500);
    lcd.clear();

    Serial.println("✓ System ready.\n");
}

// ═══════════════════════════════════════════════════
// MAIN LOOP
// ═══════════════════════════════════════════════════
void loop() {
    unsigned long now = millis();

    if (WiFi.status() != WL_CONNECTED) {
        Serial.println("⚠ WiFi lost. Reconnecting...");
        beep(3, 80);
        connectWiFi();
    }

    if (now - lastPollTime >= POLL_INTERVAL) {
        lastPollTime = now;
        pollFanStatus();
    }

    if (now - lastSensorTime >= SENSOR_INTERVAL) {
        lastSensorTime = now;
        sendSensorData();
    }
}

// ═══════════════════════════════════════════════════
// WIFI
// ═══════════════════════════════════════════════════
void connectWiFi() {
    Serial.printf("Connecting to WiFi: %s", WIFI_SSID);
    WiFi.begin(WIFI_SSID, WIFI_PASSWORD);

    int attempts = 0;
    while (WiFi.status() != WL_CONNECTED && attempts < 20) {
        delay(500);
        Serial.print(".");
        attempts++;
    }

    if (WiFi.status() == WL_CONNECTED) {
        Serial.printf("\n✓ Connected! IP: %s\n", WiFi.localIP().toString().c_str());
    } else {
        Serial.println("\n✗ WiFi failed. Will retry...");
    }
}

// ═══════════════════════════════════════════════════
// POLL FAN STATUS
// ═══════════════════════════════════════════════════
void pollFanStatus() {
    HTTPClient http;
    String url = String(SERVER_URL) + "/api/fan/status";
    http.begin(url);

    int httpCode = http.GET();

    if (httpCode == 200) {
        String payload = http.getString();
        JsonDocument doc;
        deserializeJson(doc, payload);

        const char* status = doc["status"];
        int speed = doc["speed"];

        bool newState = (String(status) == "on");

        if (newState != fanState) {
            fanState = newState;
            applyFanState();
            Serial.printf("→ Fan %s\n", fanState ? "ON" : "OFF");
            beep(fanState ? 1 : 2, 200);
        }

        if (speed != fanSpeed && fanState) {
            fanSpeed = speed;
            applyFanSpeed();
            Serial.printf("→ Speed: %d/255\n", fanSpeed);
        }
    } else {
        Serial.printf("✗ Poll failed: HTTP %d\n", httpCode);
    }

    http.end();
}

// ═══════════════════════════════════════════════════
// SEND SENSOR DATA AND UPDATE LCD
// ═══════════════════════════════════════════════════
void sendSensorData() {
    float temp = dht.readTemperature();
    float hum  = dht.readHumidity();

    if (isnan(temp) || isnan(hum)) {
        Serial.println("✗ DHT read failed");
        lcd.clear();
        lcd.setCursor(0, 0);
        lcd.print("Sensor Error");
        return;
    }

    lastTemperature = temp;
    lastHumidity = hum;
    Serial.printf("Temp: %.1f°C  Humidity: %.1f%%\n", temp, hum);

    updateLCD(temp, hum);

    JsonDocument doc;
    doc["device_identifier"] = DEVICE_ID;
    JsonArray readings = doc["readings"].to<JsonArray>();

    JsonObject t = readings.add<JsonObject>();
    t["sensor_type"] = "temperature";
    t["value"] = temp;
    t["unit"] = "°C";

    JsonObject h = readings.add<JsonObject>();
    h["sensor_type"] = "humidity";
    h["value"] = hum;
    h["unit"] = "%";

    String body;
    serializeJson(doc, body);

    HTTPClient http;
    http.begin(String(SERVER_URL) + "/api/sensor-data");
    http.addHeader("Content-Type", "application/json");
    http.addHeader("Accept", "application/json");

    int httpCode = http.POST(body);
    if (httpCode == 201) {
        Serial.println("✓ Sensor data sent");
    } else {
        Serial.printf("✗ Sensor POST failed: HTTP %d\n", httpCode);
    }
    http.end();
}

// ═══════════════════════════════════════════════════
// LCD UPDATE
// ═══════════════════════════════════════════════════
void updateLCD(float temp, float hum) {
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Temp: ");
    lcd.print(temp, 1);
    lcd.print((char)223); // degree symbol
    lcd.print("C");

    lcd.setCursor(0, 1);
    lcd.print("Hum: ");
    lcd.print(hum, 1);
    lcd.print("%");
}

// ═══════════════════════════════════════════════════
// MOTOR CONTROL
// ═══════════════════════════════════════════════════
void applyFanState() {
    if (fanState) {
        digitalWrite(MOTOR_IN2, HIGH);
        applyFanSpeed();
    } else {
        digitalWrite(MOTOR_IN2, LOW);
        ledcWrite(MOTOR_ENA, 0);
    }
}

void applyFanSpeed() {
    if (fanState) {
        ledcWrite(MOTOR_ENA, fanSpeed);
    }
}

// ═══════════════════════════════════════════════════
// AUDIO FEEDBACK
// ═══════════════════════════════════════════════════
void beep(int times, int duration) {
    for (int i = 0; i < times; i++) {
        digitalWrite(BUZZER_PIN, HIGH);
        delay(duration);
        digitalWrite(BUZZER_PIN, LOW);
        if (i < times - 1) delay(100);
    }
}

// ═══════════════════════════════════════════════════
// I2C SCANNER (uncomment in setup to use)
// ═══════════════════════════════════════════════════
void scanI2C() {
    byte error, address;
    int nDevices = 0;
    Serial.println("Scanning I2C devices...");
    for (address = 1; address < 127; address++) {
        Wire.beginTransmission(address);
        error = Wire.endTransmission();
        if (error == 0) {
            Serial.print("I2C device found at address 0x");
            if (address < 16) Serial.print("0");
            Serial.print(address, HEX);
            Serial.println("  !");
            nDevices++;
        } else if (error == 4) {
            Serial.print("Unknown error at address 0x");
            if (address < 16) Serial.print("0");
            Serial.println(address, HEX);
        }
    }
    if (nDevices == 0)
        Serial.println("No I2C devices found\n");
    else
        Serial.println("done\n");
}
